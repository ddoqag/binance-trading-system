```go
        if o == nil {
            return
        }

        o.FilledQty += event.FillQty
        o.AvgFillPrice = event.AvgPrice
        o.UpdatedAt = time.Now().UnixNano()

            o.Status = Filled
```

## ⚙️ 3. Detector（轻量但高信噪比版本）

```python
import numpy as np

class MarketRegimeDetector:
    def __init__(self):
        self.vol_window = 50

    def detect(self, features):
        """
        features:
        {
            "ret": [...],
            "ofi": [...],
            "spread": float,
            "trade_intensity": float
        }
        """

        ret = np.array(features["ret"])
        vol = np.std(ret[-self.vol_window:])

        ofi_slope = np.polyfit(range(len(features["ofi"])), features["ofi"], 1)[0]

        spread = features["spread"]
        intensity = features["trade_intensity"]

        # =========================
        # Rule-based regime engine
        # =========================

        if vol < 0.001 and abs(ofi_slope) < 0.01:
            return MarketRegime.RANGE

        if vol > 0.003 and intensity > 1.5:
            return MarketRegime.CHAOS

        if ofi_slope > 0.02:
            return MarketRegime.HIGH_VOL_TREND

        return MarketRegime.LOW_VOL_TREND
```

```python id="soft_route"
    def route(self, obs, regime):

        weights = np.array([e.weight for e in self.experts])

        # sample or weighted decision
        idx = np.random.choice(len(self.experts), p=probs)
```

```python id="softmax"
        return np.exp(x) / np.sum(np.exp(x))
```

```python id="update_weight"
class SelfEvolvingMetaAgent:

    def update(self, feedback: TradeFeedback):

        for e in self.experts:
            if e.name == feedback.expert_name:

                # =========================
                # reward shaping function
                # =========================

                reward = (
                    feedback.pnl
                    - 0.5 * feedback.risk
                    - 1.5 * feedback.drawdown
                )

                # exponential moving update
                lr = 0.05

                e.weight = e.weight * np.exp(lr * reward)

                # clamp to avoid explosion
                e.weight = np.clip(e.weight, 0.1, 10.0)

                e.pnl += feedback.pnl
                e.trades += 1
```

```python id="reproduce"
    def reproduce(self, elite):

        new_population = []

        for parent in elite:

            # keep parent (elitism)
            new_population.append(parent)

            # create offspring
            child_agent = copy.deepcopy(parent.agent)
            child_params = self.mutate(parent.params)

            child = IndividualStrategy(
                name=parent.name + "_mut",
                agent=child_agent,
                params=child_params
            )

            new_population.append(child)

        # refill population
        while len(new_population) < self.population_size:
            new_population.append(copy.deepcopy(random.choice(elite)))

        self.population = new_population
```

```python id="compile1"
        return policy
```

```python id="eval1"
    def evaluate(self, condition, context):

        feature, op, value = self.parse(condition)

        if op == ">":
            return context[feature] > float(value)
        elif op == "<":
            return context[feature] < float(value)

        return False
```

```python id="wm2"
import torch
import torch.nn as nn

class WorldModel(nn.Module):

    def __init__(self, state_dim, action_dim, hidden=128):
        super().__init__()

        self.encoder = nn.Sequential(
            nn.Linear(state_dim + action_dim, hidden),
            nn.ReLU(),
            nn.Linear(hidden, hidden)
        )

        self.dynamics = nn.GRU(hidden, hidden, batch_first=True)

        self.decoder = nn.Sequential(
            nn.Linear(hidden, hidden),
            nn.ReLU(),
            nn.Linear(hidden, state_dim)
        )
```

---

## 🧬 forward dynamics

```python id="wm3"
    def forward(self, state, action, hidden):

        x = torch.cat([state, action], dim=-1)

        h = self.encoder(x)

        out, hidden = self.dynamics(h.unsqueeze(1), hidden)

        next_state = self.decoder(out.squeeze(1))

        return next_state, hidden
```

```python id="wm4"
def imagined_rollout(world_model, agent, state, steps=50):

    hidden = None

    trajectory = []

    for _ in range(steps):

        action = agent.policy(state)

        next_state, hidden = world_model(state, action, hidden)

        trajectory.append((state, action))

        state = next_state

    return trajectory
```

def realized_vol(df):

def atr(df):

def volatility_change(df, window=5):

### 4. 波动率突破

def volatility_breakout(df):
    range_size = df["high"] - df["low"]
    return range_size > range_size.rolling(20).max().shift(1)

### 4. 量价背离

```python
def volume_price_divergence(df):
    """价格创新高但成交量未创新高"""
    price_high = df["close"] == df["close"].rolling(20).max()
    volume_high = df["volume"] >= df["volume"].rolling(20).max()
    return price_high & (~volume_high)
```

## 三、因子计算框架（完整版）

```python
import pandas as pd
import numpy as np

class VolatilityFactors:
    """波动率因子库"""

    @staticmethod
    def realized_volatility(prices, window=30):
        """已实现波动率"""
        returns = prices.pct_change()
        return returns.rolling(window).std()

    @staticmethod
    def atr(high, low, close, window=14):
        """ATR - 平均真实波幅"""
        tr1 = high - low
        tr2 = abs(high - close.shift())
        tr3 = abs(low - close.shift())
        tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
        return tr.rolling(window).mean()

    @staticmethod
    def bollinger_bands(prices, window=20, std_dev=2):
        """布林带"""
        ma = prices.rolling(window).mean()
        std = prices.rolling(window).std()
        upper = ma + std_dev * std
        lower = ma - std_dev * std
        return upper, ma, lower

    @staticmethod
    def volatility_regime(returns, window=30, threshold=0.02):
        """波动率状态（高/低）"""
        vol = returns.rolling(window).std()
        return (vol > threshold).astype(int)


class VolumeFactors:
    """成交量因子库"""

    @staticmethod
    def volume_spike(volume, window=30, threshold=2.0):
        """成交量异常"""
        avg_vol = volume.rolling(window).mean()
        return volume / avg_vol

    @staticmethod
    def volume_momentum(volume, window=10):
        """量变动量"""
        return volume.pct_change(window)

    @staticmethod
    def on_balance_volume(close, volume):
        """OBV - 能量潮"""
        obv = (np.sign(close.diff()) * volume).fillna(0).cumsum()
        return obv

    @staticmethod
    def price_volume_correlation(close, volume, window=20):
        """价量相关性"""
        returns = close.pct_change()
        vol_changes = volume.pct_change()
        return returns.rolling(window).corr(vol_changes)

    @staticmethod
    def adv_ratio(volume, window=20):
        """成交量占平均成交量比例"""
        adv = volume.rolling(window).mean()
        return volume / adv


# 综合因子引擎
class FactorEngine:
    """因子计算引擎"""

    def __init__(self):
        self.vol_factors = VolatilityFactors()
        self.vol_factors = VolumeFactors()

    def compute_all(self, df):
        """计算所有波动率和成交量因子"""
        result = pd.DataFrame(index=df.index)

        # 波动率因子
        result["rv"] = self.vol_factors.realized_volatility(df["close"])
        result["atr"] = self.vol_factors.atr(
            df["high"], df["low"], df["close"]
        )
        upper, mid, lower = self.vol_factors.bollinger_bands(df["close"])
        result["bb_upper"] = upper
        result["bb_mid"] = mid
        result["bb_lower"] = lower
        result["bb_position"] = (df["close"] - lower) / (upper - lower)

        # 成交量因子
        result["volume_spike"] = self.vol_factors.volume_spike(df["volume"])
        result["volume_mom"] = self.vol_factors.volume_momentum(df["volume"])
        result["obv"] = self.vol_factors.on_balance_volume(df["close"], df["volume"])
        result["adv_ratio"] = self.vol_factors.adv_ratio(df["volume"])

        return result
```

## 四、因子标准化

不同因子需要统一尺度。

### 1. Z-score 标准化

```python
def zscore_normalize(factor):
    """Z-score 标准化"""
    return (factor - factor.mean()) / factor.std()
```

---

### 2. Rank 标准化

```python
def rank_normalize(factor):
    """Rank 标准化 [0, 1]"""
    return factor.rank(pct=True)
```

---

### 3. 因子去极值

```python
def winsorize(factor, left=0.01, right=0.99):
    """去极值"""
    q_left = factor.quantile(left)
    q_right = factor.quantile(right)
    return factor.clip(q_left, q_right)
```

## 五、因子组合示例

```python
def create_vol_volume_alpha(df):
    """结合波动率和成交量的 Alpha"""

    # 计算因子
    vol = df["close"].pct_change().rolling(20).std()
    vol_spike = df["volume"] / df["volume"].rolling(30).mean()

    # 低波动率 + 量缩 = 可能突破
    low_vol = vol < vol.quantile(0.3)
    high_vol_spike = vol_spike > 1.5

    alpha = (low_vol.astype(int) - 0.5) * high_vol_spike

    return alpha
```
